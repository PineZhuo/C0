void main(){
    int n;
    scan(n);
    if(n < 10){
        n = n+1;
    }
    else{
        if(n <= 20){
            n = n+2;
        }
        else{
            if(n == 30)
                n = n + 3;
            else
                n = n + 4;
        }
	n= n-1;
    }
    print(n);
    return ;
}
